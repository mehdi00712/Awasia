// Firestore security rules (read-only for public, write-only for admins via Auth + custom claims if desired)
// For quick start: allow reads to products; restrict writes to authenticated users.
// For stricter control, use an allowlist by email on the server or custom claims.
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /products/{doc} {
      allow read: if true;
      allow write: if request.auth != null; // tighten later
    }
  }
}
